title_red_druk = {
	dignity = 5
	realm_in_name = yes
	show_as_title = yes
	grant_limit = 1
	opinion_effect = 5
	is_voter = yes
	is_high_prio = yes
	
	monthly_salary = 0.1
	monthly_prestige = 0.75
	
	
	is_unique = yes
	
	allowed_to_hold = {
		OR = {
			male_can_hold_minor_title_trigger = yes
			female_can_hold_minor_title_trigger = yes
			has_law = true_cognatic_succession
		}
		is_adult = yes
		prisoner = no
		is_incapable = no
		has_regent = no
		is_landed = yes
		NOT = { trait = monk }
		NOT = { trait = nun }
		is_theocracy = no
	}
	
	allowed_to_grant = {
		is_feudal = yes
		has_landed_title = e_bhutan
		primary_title = { 
			OR = {
				has_law = parliament_laws_bhutan_1
				has_law = parliament_laws_bhutan_2
				has_law = parliament_laws_bhutan_3
			}
			NOT = { 
				has_law = parliament_laws_bhutan_0
			}
		}
	}
	
	gain_effect = {
		character_event = { id = DRUK.14 }
	}
	lose_effect = {
		remove_trait=red_druk
		if = {
			limit = {
				FROM = {
					NOT = {
						has_character_flag = swapping_councillors
					}
				}
			}
			opinion = { who = FROM modifier = opinion_fired_from_council }
			opinion = { who = FROM modifier = opinion_recently_fired }
		}
		destroy_landed_title = c_red_druk
	}
	retire_effect = {
		if = {
			limit = {
				NOT = {
					age = 65
				}
				health = 3
				NOT = {
					is_sick_or_injured_trigger = yes
				}
			}
			
			FROM = { 
				opinion = { who = ROOT modifier = opinion_resigned_from_council }
			}
		}
		destroy_landed_title = c_red_druk
	}

	message = yes
}

title_chechnik = {
	dignity = 3
	realm_in_name = yes
	grant_limit = 6
	is_voter = yes
	is_high_prio = yes
	
	monthly_salary = 0.075
	monthly_prestige = 0.5
	

	
	allowed_to_hold = {
		OR = {
			male_can_hold_minor_title_trigger = yes
			female_can_hold_minor_title_trigger = yes
			has_law = true_cognatic_succession
		}
		is_adult = yes
		prisoner = no
		is_incapable = no
		has_regent = no
		NOT = { trait = monk }
		NOT = { trait = nun }
	}
	
	allowed_to_grant = {
		OR = {
			has_landed_title = k_israel
			has_landed_title = e_zion
		}
		religion_group = jewish_group
		primary_title = {
			NOT = { 
				has_law = parliament_laws_israel_0
			}		
			OR = {
				has_law = parliament_laws_israel_1 
				has_law = parliament_laws_israel_2
				has_law = parliament_laws_israel_3
			}
		}
	}
	
	gain_effect = {
		character_event = { id = DRUK.3 }
	}
	lose_effect = {
		character_event = { id = DRUK.4 }
		if = {
			limit = {
				FROM = {
					NOT = {
						has_character_flag = swapping_councillors
					}
				}
			}
			opinion = { who = FROM modifier = opinion_fired_from_council }
			opinion = { who = FROM modifier = opinion_recently_fired }
		}
	}

	message = yes
}
