# should_evolve_cultures_trigger
# non_historical_culture_names_in_alt_start_trigger
# religions_are_standard_trigger

should_evolve_cultures_trigger = {
	hidden_trigger = {
		trigger_if = {
			limit = { is_alternate_start = yes }

			has_alternate_start_parameter = {
				key = culture_names
				value = historical
			}
		}
	}
}

non_historical_culture_names_in_alt_start_trigger = {
	hidden_trigger = {
		is_alternate_start = yes

		NOT = {
			has_alternate_start_parameter = {
				key = culture_names
				value = historical
			}
		}
	}
}

religions_are_standard_trigger = {
	hidden_trigger = {
		trigger_if  = {
			limit = { is_alternate_start = yes }

			has_alternate_start_parameter = {
				key = religion_names
				value = historical
			}
		}
	}
}
