obj_zarist_emulate_peter = {
	type = character
	can_cancel = no
	potential = {
		OR = {
			religion = zarist
			religion = petrist
		}
	}
	
	allow = {
		is_adult = yes
		NOT = { trait = incapable }
		NOT = { zarist_emulated_trigger = yes }
	}
	chance = {
		factor = 50
		
		modifier = {
			factor = 2
			trait = zealous
		}
		modifier = {
			factor = 0.2
			trait = cynical
		}
		modifier = {
			factor = 0.5
			is_unimportant_trigger = yes
		}
	}
	creation_effect = {
		character_event = { id = zarist.100 }
	}
	success = {
		custom_tooltip = {
			text = zarist_emulate_tt
			has_character_modifier = zarist_emulated_peter
		}
	}
	abort_effect = {
		hidden_tooltip = {
			clr_character_flag = zarist_emulate_becoming
		}
	}
	effect = {
		tooltip = {
			piety = 50
			prestige = 50
		}
	}
}
