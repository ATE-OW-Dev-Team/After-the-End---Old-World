european_unification = {
    name = CB_NAME_EUROPEAN_UNIFICATION
	war_name = WAR_NAME_EUROPEAN_UNIFICATION
	sprite = 17
	truce_days = 3650
	is_permanent = yes
	check_de_jure_tier = KING # this scans all dejure kingdoms for the (counties or duchies? idk) which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	allowed_to_target_tributaries = no
	
	sort_priority = 1900

#	can_use_gui = {
#
#	}

	can_use = {
		ROOT = {
			has_landed_title = e_european_union
			independent = yes
			NOT = { same_realm = FROM }
			mercenary = no
			NOR = {
				religion = stalinist
				religion = trotskyite
			}
		}
	}

	can_use_title = {
		NOT = { de_jure_liege_or_above = e_britannia }
		capital_scope = {
			region = world_europe
		}
	}
	
	is_valid = {
		ROOT = {
			has_landed_title = e_european_union
			independent = yes
			NOT = { same_realm = FROM }
			mercenary = no
			NOR = {
				religion = stalinist
				religion = trotskyite
			}
		}
	}
	on_add = {
		if = {
			limit = { defender = { is_offmap_governor = offmap_china } }
			attacker = {  
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes 
			}
		}
	}
	
	on_success = {
		ROOT = {
			participation_scaled_prestige = 100
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 100
			}
		}
		
		FROM = { prestige = -100 }
	}

	on_success_title = {
		custom_tooltip = { 
			text = other_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
						type = invasion
					}
				}
			}
		}
		if = {
			limit = {
				holder_scope = { character = FROM }
			}
			usurp_title_only = { target = ROOT type = invasion }
		}
	}

	on_fail = {
		ROOT = { prestige = -150 }
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 10.0
			}
			ROOT = { prestige = -200 }
		}
		FROM = {
			participation_scaled_piety = 100
			participation_scaled_prestige = 200
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 100
				participation_scaled_prestige = 200
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
european_invasion = {
    name = CB_NAME_EUROPEAN_INVASION
	war_name = WAR_NAME_EUROPEAN_INVASION
	sprite = 17
	truce_days = 36500
	is_permanent = yes
	check_de_jure_tier = EMPEROR
	allowed_to_target_tributaries = yes
	
	sort_priority = 2000

	can_use_gui = {
		ROOT = {
			prestige = 2000
			wealth = 1000
		}
	}

	can_use = {
		ROOT = {
			has_landed_title = e_european_union
			independent = yes
			NOT = { same_realm = FROM }
			mercenary = no
		}
		hidden_tooltip = {
		    NOT = { has_global_flag = europe_conquers_britain }
		}
	}

	can_use_title = {
	    title = e_britannia		
	}
	
	is_valid = {
		ROOT = {
			has_landed_title = e_european_union
			independent = yes
			NOT = { same_realm = FROM }
			mercenary = no
		}
	}
	on_add = {
		ROOT = {
			wealth = -1000
		}
		if = {
			limit = {
			    NOT = { has_global_flag = europe_invades_britain }
			}
		    ROOT = {
			    narrative_event = { id = european_union_event.002 }
			}
		}
		#else = {
		#	ROOT = {
		#	    narrative_event = { id = european_union_event.004 }
		#	}
		#}
		if = {
			limit = { defender = { is_offmap_governor = offmap_china } }
			attacker = {  
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes 
			}
		}
	}
	
	on_add_title = {
		any_independent_ruler = {
			limit = {
				mercenary = no
				NOT = { character = FROM }
				#religion = FROM
				#NOT = { war_with = FROM }
				NOT = { war_with = ROOT }
				any_realm_title = {
					de_jure_liege_or_above = PREVPREV
				}
			}
			join_defender_wars = FROM
		}
	}
	
	on_success = {
		ROOT = {
			narrative_event = { id = european_union_event.003 }
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 1000
			}
		}
		
		FROM = { prestige = -4000 }
	}

	on_success_title = {
		hidden_tooltip = {
			ROOT = {
				vassalize_or_take_under_title = {
					title = PREV
					enemy = FROM
					type = invasion
				}
			}
			any_independent_ruler = {
				limit = {
					mercenary = no
					#religion = FROM
					any_realm_title = {
					    de_jure_liege_or_above = PREVPREV
					}
				}
				show_scope_change = no
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREVPREV
						enemy = PREV
						type = invasion
					}
				}
			}
		}
		if = {
			limit = {
				holder_scope = { character = FROM }
			}
			usurp_title_only = { target = ROOT type = invasion }
		}
	}

	on_fail = {
		ROOT = { prestige = -8000 }
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 40.0
			}
			ROOT = { prestige = -8000 }
		}
		FROM = {
			participation_scaled_piety = 1000
			participation_scaled_prestige = 2000
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 500
				participation_scaled_prestige = 1000
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
