choose_the_right_caliph = {
	name = CB_NAME_CHOOSE_THE_RIGHT_CALIPH
	war_name = WAR_NAME_CHOOSE_THE_RIGHT_CALIPH
	sprite = 11
	truce_days = 3650
	is_permanent = yes
	can_ask_to_join_war = yes
	
	can_use = {
		ROOT = {
			religion = FROM
			religion_group = muslim
			OR = {
				AND = {
					is_anti_caliph = yes
					FROM = { controls_religion = yes }
				}
				AND = {
					controls_religion = yes
					FROM = { is_anti_caliph = yes }
				}
			}
		}
	}
	
	is_valid = {
		ROOT = {
			religion = FROM
			religion_group = muslim
			OR = {
				AND = {
					is_anti_caliph = yes
					FROM = { controls_religion = yes }
				}
				AND = {
					controls_religion = yes
					FROM = { is_anti_caliph = yes }
				}
			}
		}
	}
	
	on_success = {
		any_attacker = {
			show_scope_change = no
			limit = { character = ROOT }
			participation_scaled_piety = 150
			participation_scaled_prestige = 250
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_piety = 150
				participation_scaled_prestige = 250
			}
		}
		
		ROOT = {
			show_scope_change = no
			religion_authority = {
				modifier = ended_caliph_conflict
				years = 25
			}
		}
		
		FROM = {
			show_scope_change = no
			piety = -150
			prestige = -250
			set_character_flag = failed_caliph
			if = {
				limit = {
					is_anti_caliph = yes
				}
				end_anti_caliphate = yes
			}
			if = {
				limit = {
					controls_religion = yes
				}
				if = {
					limit = { religion = sunni }
					d_sunni = { usurp_title = ROOT }
				}
				if = {
					limit = { religion = ibadi }
					d_ibadi = { usurp_title = ROOT }
				}
				if = {
					limit = { religion = shiite }
					d_shiite = { usurp_title = ROOT }
				}
				if = {
					limit = { religion = bahai }
					d_bahai = { usurp_title = ROOT }
				}
				if = {
					limit = { religion = wahhabi }
					d_wahhabi = { usurp_title = ROOT }
				}
			}
		}
	}

	on_fail = {
		ROOT = {
			show_scope_change = no
			piety = -75
			prestige = -125
			if = {
				limit = {
					contros_religion = yes
				}
				religion_authority = {
					modifier = caliph_conflict_continue
					years = 30
				}
			}
		}
		FROM = {
			show_scope_change = no
			piety = 75
			prestige = 125
		}
	}

	on_reverse_demand = {
		any_defender = {
			show_scope_change = no
			limit = { character = FROM }
			participation_scaled_piety = 200
			participation_scaled_prestige = 300
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 200
				participation_scaled_prestige = 300
			}
		}
		FROM = {
			show_scope_change = no
			if = {
				limit = { controls_religion = yes }
				religion_authority = {
					modifier = ended_caliph_conflict
					years = 30
				}
			}
		}
		ROOT = {
			show_scope_change = no
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			piety = -200
			prestige = -300
			if = {
				limit = {
					controls_religion = yes
				}
				religion_authority = {
					modifier = caliph_conflict_continue
					years = 30
				}
			}
			if = {
				limit = {
					is_anti_caliph = yes
				}
				set_character_flag = failed_caliph
				end_anti_caliphate = yes
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
