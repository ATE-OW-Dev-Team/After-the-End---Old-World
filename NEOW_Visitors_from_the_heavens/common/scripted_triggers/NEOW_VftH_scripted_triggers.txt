######################################
# Scripted Triggers for determine exile
######################################

is_alien_religion_trigger = {
	OR = {
		religion_group = alien_group
		religion_group = synth_group
	}
}

is_alien_culture_trigger = {
	OR = {
		culture_group = blorgian
		culture_group = droid
		culture_group = taralonian
	}
}

has_alien_trait_trigger = {
	OR = {
		trait = blorg
		trait = android
		trait = pasharti
		trait = jeferi
	}
}
