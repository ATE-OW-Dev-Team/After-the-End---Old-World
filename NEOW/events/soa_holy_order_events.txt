# The Grand Master demands one of your sons due to debt
character_event = {
	id = SoA.1000
	hide_window = yes
	
	min_age = 16
	only_playable = yes
	religion_group = christian
	
	trigger = {
		OR = {
			religion = catholic
			religion = cathar
			religion = fraticelli
			religion = waldensian
			religion = lollard
		}
		is_heretic = no
		has_character_modifier = borrowed_from_knights_templar
		NOT = { has_character_flag = sent_son_to_knights_templar }
		any_child = {
			is_female = no
			is_adult = yes
			is_ruler = no
			prisoner = no
			NOT = { trait = incapable }
			ROOT = {
				NOT = { 
					current_heir = { 
						character = PREVPREV 
					}
				}
			}
		}
		has_dlc = "Sons of Abraham"
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	immediate = {
		random_child = {
			limit = {
				is_female = no
				is_adult = yes
				is_ruler = no
				prisoner = no
				NOT = { trait = incapable }
				ROOT = {
					NOT = { 
						current_heir = { 
							character = PREVPREV 
						}
					}
				}
			}
			character_event = {
				id = SoA.1001
			}
		}
	}
}
