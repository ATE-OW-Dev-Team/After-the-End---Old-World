#[on_combat_pulse] Druidic character becomes living legend
character_event = {
	id = celtic.11
	desc = celtic.11.desc
	picture = GFX_evt_headhunter
	border = GFX_event_normal_frame_war

	hide_from = yes
	
	is_triggered_only = yes

	min_age = 16
	capable_only = yes
	prisoner = no
	
	trigger = {
		is_alive = yes
		trait = head1
		trait = brave
		OR = {
			trait = tough_soldier
			trait = skilled_tactician
			trait = brilliant_strategist
			trait = duelist
		}
		NOT = { trait = head2 }
		religion_openly_celtic_or_reformed_trigger = yes
		NOT = {
			has_religion_feature = religion_peaceful
		}
		had_character_flag = { flag = ancrel_headhunter days = 780 }
	}
	
	weight_multiplier = {
		days = 100
		modifier = {
			factor = 2
			trait = brave
		}		
		modifier = {
			factor = 2
			martial = 13
		}
		modifier = {
			factor = 2
			is_strong_trigger = yes
		}
		modifier = {
			factor = 2
			trait = genius
		} # \/ These two /\ stack, so genius gets x10

			factor = 0.1
