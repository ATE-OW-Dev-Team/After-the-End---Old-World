	desc = EVTDESC_RIP_11506
	picture = GFX_evt_lunatic

	is_triggered_only = yes # on_bi_yearly_pulse

	ai = no

		trait = one_eyed
		trait = one_handed
		trait = one_legged

		name = EVTOPTA_RIP_11506

	}
}

# Crazy cannibal eat character and gain traits
character_event = {
	id = RIP.11510
	desc = EVTDESC_RIP_11510
	picture = GFX_evt_impalement

	is_triggered_only = yes # execute_eat, rip_torture_decisions.txt

	option = {
		name = EVTOPTA_RIP_11510

		### Congenital traits
		if = {
			limit = {
				FROM = { trait = genius }
				NOT = { trait = genius }
			}

			if = {
				limit = { trait = imbecile }
				remove_trait = imbecile
			}

			if = {
				limit = { trait = slow }
				remove_trait = slow
			}

			if = {
				limit = { trait = dull }
				remove_trait = dull
			}

			if = {
				limit = { trait = quick }
				remove_trait = quick
			}

			add_trait = genius
		}
		else_if = {
			limit = {
				FROM = { trait = quick }

				NOR = {
					trait = genius
					trait = quick
				}
			}

			if = {
				limit = { trait = imbecile }
				remove_trait = imbecile
			}

			if = {
				limit = { trait = slow }
				remove_trait = slow
			}

			if = {
				limit = { trait = dull }
				remove_trait = dull
			}

			add_trait = quick
		}

		if = {
			limit = {
				FROM = { trait = strong }
				NOT = { trait = strong }
			}

			if = {
				limit = { trait = weak }
				remove_trait = weak
			}

			if = {
				limit = {
					trait = feeble
				}
				remove_trait = feeble
			}

			add_trait = strong
		}

		if = {
			limit = {
				FROM = {
					trait = fair
				}
				NOT = { trait = fair }
			}

			if = {
				limit = { trait = ugly }
				remove_trait = ugly
			}
			add_trait = fair
		}

		### Personality traits - Sins and virtues
		if = {
			limit = {
				FROM = { trait = chaste }
				NOT = { trait = chaste }
			}
			add_trait = chaste
		}
		else_if = {
			limit = {
				FROM = { trait = lustful }
				NOT = { trait = lustful }
			}
			add_trait = lustful
		}

		if = {
			limit = {
				FROM = { trait = temperate }
				NOT = { trait = temperate }
			}
			add_trait = temperate
			lose_weight_medium_effect = yes
			weight_trait_check_effect = yes
		}
		else_if = {
			limit = {
				FROM = { trait = gluttonous }
				NOT = { trait = gluttonous }
			}
			add_trait = gluttonous
			gain_weight_medium_effect = yes
			weight_trait_check_effect = yes
		}

		if = {
			limit = {
				FROM = { trait = charitable }
				NOT = { trait = charitable }
			}
			add_trait = charitable
		}
		else_if = {
			limit = {
				FROM = { trait = greedy }
				NOT = { trait = greedy }
			}
			add_trait = greedy
		}

		if = {
			limit = {
				FROM = { trait = diligent }
				NOT = { trait = diligent }
			}
			add_trait = diligent
		}
		else_if = {
			limit = {
				FROM = { trait = slothful }
				NOT = { trait = slothful }
			}
			add_trait = slothful
		}

		if = {
			limit = {
				FROM = { trait = patient }
				NOT = { trait = patient }
			}
			add_trait = patient
		}
		else_if = {
			limit = {
				FROM = { trait = wroth }
				NOT = { trait = wroth }
			}
			add_trait = wroth
		}

		if = {
			limit = {
				FROM = { trait = kind }
				NOT = { trait = kind }
			}
			add_trait = kind
		}
		else_if = {
			limit = {
				FROM = { trait = envious }
				NOT = { trait = envious }
			}
			add_trait = envious
		}

		if = {
			limit = {
				FROM = { trait = humble }
				NOT = { trait = humble }
			}
			add_trait = humble
		}
		else_if = {
			limit = {
				FROM = { trait = proud }
				NOT = { trait = proud }
			}
			add_trait = proud
		}

		### Personality traits - Other
		if = {
			limit = {
				FROM = { trait = deceitful }
				NOT = { trait = deceitful }
			}
			add_trait = deceitful
		}
		else_if = {
			limit = {
				FROM = { trait = honest }
				NOT = { trait = honest }
			}
			add_trait = honest
		}

		if = {
			limit = {
				FROM = { trait = craven }

				NOR = {
					trait = craven
					trait = berserker
				}
			}
			add_trait = craven
		}
		else_if = {
			limit = {
				FROM = { trait = brave }
				NOT = { trait = brave }
			}
			add_trait = brave
		}

		if = {
			limit = {
				FROM = { trait = shy }
				NOT = { trait = shy }
			}
			add_trait = shy
		}
		else_if = {
			limit = {
				FROM = { trait = gregarious }
				NOT = { trait = gregarious }
			}
			add_trait = gregarious
		}

		if = {
			limit = {
				FROM = { trait = ambitious }
				NOT = { trait = ambitious }
			}
			add_trait = ambitious
		}
		else_if = {
			limit = {
				FROM = { trait = content }
				NOT = { trait = content }
			}
			add_trait = content
		}

		if = {
			limit = {
				FROM = { trait = arbitrary }
				NOT = { trait = arbitrary }
			}
			add_trait = arbitrary
		}
		else_if = {
			limit = {
				FROM = { trait = just }
				NOT = { trait = just }
			}
			add_trait = just
		}

		if = {
			limit = {
				FROM = { trait = cynical }
				NOT = { trait = cynical }
			}
			add_trait = cynical
		}
		else_if = {
			limit = {
				FROM = { trait = zealous }
				NOT = { trait = zealous }
			}
			add_trait = zealous
		}

		if = {
			limit = {
				FROM = { trait = paranoid }
				NOT = { trait = paranoid }
			}
			add_trait = paranoid
		}
		else_if = {
			limit = {
				FROM = { trait = trusting }
				NOT = { trait = trusting }
			}
			add_trait = trusting
		}

		if = {
			limit = {
				FROM = { trait = cruel }
				NOT = { trait = cruel }
			}
			add_trait = cruel
		}

		if = {
			limit = {
				FROM = { trait = erudite }
				NOT = { trait = erudite }
			}
			add_trait = erudite
		}

		if = {
			limit = {
				FROM = { trait = stubborn }
				NOT = { trait = stubborn }
			}
			add_trait = stubborn
		}

		### Other traits
		if = {
			limit = {
				FROM = { immortal = yes }
				immortal = no
			}

			add_trait = immortal

			if = {
				limit = { trait = incapable }
				remove_trait = incapable
			}

			if = {
				limit = { has_dlc = "Holy Fury" }

				if = {
					limit = { is_female = no }

					create_bloodline = {
						type = immortal_bloodline_01
					}
				}
				else = {
					create_bloodline = {
						type = immortal_bloodline_01
						inheritance = matrilineal
					}
				}
			}
		}

		### Controls religion
		if = {
			limit = {
				has_character_flag = pope_eater
				NOT = { trait = sympathy_christendom }
			}

			remove_trait_silently_zealous_effect = yes
			add_trait = sympathy_christendom
		}
		else_if = {
			limit = {
				has_character_flag = caliph_eater
				NOT = { trait = sympathy_islam }
			}

			remove_trait_silently_zealous_effect = yes
			add_trait = sympathy_islam
		}
		else_if = {
			limit = {
				has_character_flag = fylkir_feast
				NOT = { trait = sympathy_pagans }
			}

			remove_trait_silently_zealous_effect = yes
			add_trait = sympathy_pagans
		}
		else_if = {
			limit = {
				has_character_flag = fire_eater
				NOT = { trait = sympathy_zoroastrianism }
			}

			remove_trait_silently_zealous_effect = yes
			add_trait = sympathy_zoroastrianism
		}
		else_if = {
			limit = {
				has_character_flag = non_kosher
				NOT = { trait = sympathy_judaism }
			}

			remove_trait_silently_zealous_effect = yes
			add_trait = sympathy_judaism
		}
		else_if = {
			limit = {
				has_character_flag = dont_even_exist
				NOT = { trait = sympathy_indian }
			}

			remove_trait_silently_zealous_effect = yes
			add_trait = sympathy_indian
		}

		clr_character_flag = pope_eater
		clr_character_flag = caliph_eater
		clr_character_flag = fylkir_feast
		clr_character_flag = fire_eater
		clr_character_flag = non_kosher
		clr_character_flag = dont_even_exist
	}
}

	is_triggered_only = yes # execute_eat, rip_torture_decisions.txt

		name = EVTOPTA_RIP_11511

		add_rival = FROMFROM

		opinion = {
			name = opinion_killed_close_kin
			who = FROMFROM
			years = 5
		}
	}
}
