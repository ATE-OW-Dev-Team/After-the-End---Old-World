# The Grand Master humbly requests the right to build a castle in your demesne
character_event = {
	id = celtic.1324	
	hide_window = yes
	
	only_playable = yes
	has_dlc = "Sons of Abraham"
	
	trigger = {
		religion = celtic_pagan_reformed
		is_heretic = no
		
		d_red_dragons = {
			holder_scope = {
				wealth = 300
				NOT = {
					realm_size = 10
				}
				NOT = {
					num_of_count_titles = 1
				}
			}
		}
		
		any_demesne_province = {
			has_empty_holding = yes
			empire = {
				OR = {
					title = e_france
					title = e_hre
					title = e_spain
					title = e_britannia
				}
			}
		}
		
		NOT = { has_character_modifier = expelled_d_red_dragons }
		NOT = {
			any_liege = {
				has_character_modifier = expelled_d_red_dragons
			}
		}
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			d_red_dragons = {
				holder_scope = {
					NOT = { demesne_size = 1 }
				}
			}
		}
		modifier = {
			factor = 0.5
			d_red_dragons = {
				holder_scope = {
					NOT = { demesne_size = 2 }
				}
			}
		}
		modifier = {
			factor = 2.0
			d_red_dragons = {
				holder_scope = {
					realm_size = 5
				}
			}
		}
	}
	
	immediate = {
		random_demesne_province = {
			limit = {
				has_empty_holding = yes
				empire = {
					OR = {
						title = e_france
						title = e_hre
						title = e_spain
						title = e_britannia
					}
				}
			}
			province_event = {
				id = celtic.1325
			}
		}
	}
}
