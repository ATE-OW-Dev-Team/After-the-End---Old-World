character_event = {
	id = ECOINF.1003
	desc = EVTDESC_ECOINF.1003
	picture = "GFX_evt_diplomat_hf"
	
	hide_window = yes
	trigger = {
		independent = yes
		in_revolt = no
		rebel = no
		any_realm_province = {
			distance = {
				who = ROOT
				value > 250
			}
			NOT = {
				has_province_modifier = distant_from_capital
			}
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		
		random_realm_province = {
			limit = {
				distance = {
					who = ROOT
					value > 250
				}
				NOT = {
					has_province_modifier = distant_from_capital
				}
			}

			add_province_modifier = { 
				name = distant_from_capital 
				duration = 730 
				hidden = yes
			}
		}
	}
	
	option = {
		name = "OK"
	}
}
