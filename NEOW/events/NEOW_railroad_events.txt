namespace = railroad

narrative_event = {
	id = railroad.1000
	
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {	
		k_soitoskan = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_khorasan
			}
		}
		k_khorasan = {
			grant_title = ROOT
			copy_title_laws = k_soitoskan
			copy_title_history = k_soitoskan
			set_name=""
			adjective=""
		}
		k_soitoskan = {
			destroy_landed_title = THIS
		}
		activate_title = { title = k_soitoskan status = no }
		
		k_khorasan = {
			if = {
				limit = {
					NOT = {
						has_dlc = "Zeus"
						OR = {
							has_law = crown_authority_0
							has_law = crown_authority_1
							has_law = crown_authority_2
							has_law = crown_authority_3
							has_law = crown_authority_4
						}
					}
				}
				add_law = crown_authority_0
			}
		}
		any_playable_ruler = {
			limit = {
				NOT = { has_landed_title = k_khorasan }
				ai = no
			}
			narrative_event = { id = railroad.1001 }
		}
		
		any_realm_province = {
			limit = {
				kingdom = {
					title = k_khorasan
				}
			}
			score_value = {
				value = 1
			}
			count = 5
			culture = soitoskan
			religion = ROOT
		}
		
		# The Hungarian culture in the Ukraine is lost
		while = {
			limit = {
				check_variable = {
					which = soitoskan_province_leave
					value >= 1
				}
			}
			subtract_variable = {
				which = soitoskan_province_leave
				value = 1
			}
			random_realm_province = {
				limit = {
					culture = soitoskan
					empire = {
						title = e_steppes
					}
					any_neighbor_province = {
						NOT = { culture = soitoskan }
					}
				}
				random_neighbor_province = {
					limit = {
						NOT = { culture = soitoskan }
					}
					reverse_culture = PREV
					reverse_religion = PREV
				}
			}
		}
		# Lose control of the Ukraininan lands
		any_realm_province = {
			limit = {
				kingdom = {
					title = e_steppes
				}
				owner = {
					# Human vassals get to keep their stuff
					OR = {
						ai = yes
						character = ROOT
					}
				}
			}
			if = {
				limit = { 
					duchy = {
						holder_scope = {
							OR = {
								ai = yes
								character = ROOT
							}
							OR = {
								same_realm = ROOT
								character = ROOT
							}
						}
					}
				}
				duchy = { destroy_landed_title = THIS }
			}
			county = {
				create_character = {
					random_traits = yes
					dynasty = none
					religion = PREV
					culture = PREV
					female = no
					age = 25
					trait = peasant_leader
				}
				new_character = {
					usurp_title_plus_barony_if_unlanded = PREV
					set_defacto_liege = THIS
				}
			}
		}
		
		any_claim = {
			remove_claim = ROOT
		}
		
		narrative_event = { id = railroad.1002 }
	}
}
