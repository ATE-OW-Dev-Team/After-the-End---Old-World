# request to join the order (courtier)
character_event = {
	id = ancrel.9100
	hide_window = yes # only shows for ai
	
	only_men = yes
	min_age = 16
	max_age = 45
	only_capable = yes
	prisoner = no
	has_dlc = "Sons of Abraham"
	
	trigger = {
		is_ruler = no
		is_heretic = no
		is_councillor = no
		
		martial = 5
		
		OR = {
			father_even_if_dead = {
				primary_title = { higher_tier_than = BARON }
			}
			mother_even_if_dead = {
				primary_title = { higher_tier_than = BARON }
			}
		}
		
		any_liege = {
			holy_order = no
		}
		
		any_mercenary_band = {
			is_title_active = THIS
			holy_order = yes
			holder_scope = { religion = ROOT }
			ROOT = {
				NOR = {
					has_character_flag = expelled_@PREV
					any_liege = {
						has_character_flag = expelled_@PREVPREV
					}
				}
			}
		}

		NOT = {
			any_spouse = {
				OR = {
					is_landed = yes
					AND = { # Won't happen if spouse is pregnant with non-bastard child
						is_pregnant = yes
						father_of_unborn = { character = ROOT } 
					}
				}
			}
		}
	}
	
	immediate = {
		random_character = {
			limit = {
				primary_title = {
					holy_order = yes
					holder_scope = { religion = ROOT }
					ROOT = {
						NOR = {
							has_character_flag = expelled_@PREV
							any_liege = {
								has_character_flag = expelled_@PREVPREV
							}
						}
					}
				}
			}

			primary_title = { save_event_target_as = holy_order }
		}
	}

	mean_time_to_happen = {
		years = 100
		modifier = {
			factor = 0.25
			trait = zealous
		}
		modifier = {
			factor = 2.0
			trait = celibate
		}
		modifier = {
			factor = 0.5
			trait = homosexual
		}
		modifier = {
			factor = 0.7
			trait = content
		}
		modifier = {
			factor = 1.25
			trait = chaste
		}
		modifier = {
			factor = 0.75
			NOT = { age = 25 }
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 0.5
			trait = lustful
		}
		modifier = {
			factor = 2.0
			trait = cynical
		}
		modifier = {
			factor = 0.5
			trait = hedonist
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			any_heir_title = {
				always = yes
			}
		}
		modifier = {
			factor = 2.0
			is_primary_heir = yes
		}
	}
	
	option = { # Ask to go
		name = OK
		liege = {
			character_event = {
				id = ancrel.9101
			}
		}
	}
	option = { # It's a bad idea
		name = OK
	}
}

			}
		}

		any_child = {
			OR = {

					}

			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	immediate = {
		random_character = {
			limit = {
				religion = ROOT
				primary_title = {
					is_banking_holy_order_trigger = yes
					ROOT = {
						has_character_flag = borrowed_from_@PREV
						NOT = { has_character_flag = sent_child_to_@PREV}
					}
				}
			}

			primary_title = { save_event_target_as = holy_order }
		}
		random_child = {
			limit = {
				OR = {
					is_female = no
					is_feminist_religion_trigger = yes
				}
				is_adult = yes

character_event = {
	id = ancrel.9121
	
	hide_window = yes
	
	min_age = 16
	only_playable = yes
	has_dlc = "Sons of Abraham"
	
	trigger = {
		any_mercenary_band = {
			is_banking_holy_order_trigger = yes
			holder_scope = {
				religion = ROOT
				wealth = 300
			}
			ROOT = {
				is_heretic = no
				has_character_flag = borrowed_from_@PREV
				NOT = { has_character_flag = sent_son_to_@PREV }
				NOT = { has_character_flag = expelled_@PREV }
				NOT = {
					any_liege = {
						has_character_flag = expelled_@PREV
					}
				}
			}
		}
		
		any_demesne_province = { has_empty_holding = yes }
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	immediate = {
		random_character = {
			limit = {
				primary_title = {
					is_banking_holy_order_trigger = yes 
					ROOT = {
						has_character_flag = borrowed_from_@PREV
						NOT = { has_character_flag = sent_son_to_@PREV }
						NOT = { has_character_flag = expelled_@PREV }
						NOT = {
							any_liege = {
								has_character_flag = expelled_@PREV
							}
						}
					}
				}
				religion = ROOT
				wealth = 300
			}

			primary_title = {
				save_event_target_as = holy_order			
			}
		}

			limit = {
				has_empty_holding = yes
			}
			save_event_target_as = target_land

				id = ancrel.9122
			}
		}

	}
