				has_minor_title = title_appointed_regent
			}
		}
		any_courtier_or_vassal = {
			OR = {
				is_main_spouse = ROOT
				is_friend = ROOT
				sibling = ROOT
				is_parent_of = ROOT
			}
		}
	}
	mean_time_to_happen = {
		days = 0
	}
	immediate = {
		random_courtier_or_vassal = {
			limit = {
				can_hold_title = title_appointed_regent
				OR = {
					is_main_spouse = ROOT
					is_friend = ROOT
					sibling = ROOT
					is_parent_of = ROOT
				}
			}
