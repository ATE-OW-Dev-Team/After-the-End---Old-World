		months = 1300

			factor = 0.5

			learning = 7
		}

			factor = 0.85
			learning = 15
		}

			factor = 0.75
			learning = 15
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 5 }
		}

			NOT = { learning = 1 }
		}

			OR = {
				has_game_rule = {

				}
				has_game_rule = {
					name = culture_conversion

				}
			}
		}
	}

	option = {
		name = EVTOPTAMPE.15
