#Rumeli
character_event = {
	id = MPE.15
	desc = "EVTDESCMPE.15"
	picture = GFX_evt_emissary
	
	only_playable = yes
	
	trigger = {
		OR = {
			culture = turkish
			culture = elbistanli
			culture = egeli
			culture = kapadokyali
			culture = karadenizci
		}
		any_demesne_province = {
			OR = {
				culture = greek
				culture = makedonas
				culture = aigionoi
			}
			any_province_holding = {
				NOT = { holding_type = nomad }
			}
		}
		NOT = { 
			controls_religion = yes
		}
	}
	
	mean_time_to_happen = {
		months = 1300
		modifier = {
			factor = 0.5
			culture = rumeli
		}
		modifier = {
			factor = 0.8
			learning = 7
		}
		modifier = {
			factor = 0.85
			learning = 15
		}
		modifier = {
			factor = 0.75
			learning = 15
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 2.0
			NOT = { learning = 1 }
		}
		modifier = {
			factor = 0.65
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTAMPE.15
		random_demesne_province = {
			limit = {
				OR = {
					culture = greek
					culture = makedonas
					culture = aigionoi
				}
				any_province_holding = {
					NOT = { holding_type = nomad }
				}
			}
			culture = rumeli
		}
	}
}
