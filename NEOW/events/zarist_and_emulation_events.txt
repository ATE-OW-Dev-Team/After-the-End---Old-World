					subtract_variable = { which = emulation_progress value = 1 }
				}
				if = {
					limit = { trait = ambitious }
					subtract_variable = { which = emulation_progress value = 1 }
				}
			}
		}
		subtract_variable = { which = local_emulation_progress which = emulation_progress }
		if = {
			limit = { has_character_flag = zarist_emulate_becoming }
			if = {
				limit = { check_variable = { which = emulation_progress value = 3 } }
				change_variable = { which = emulation_years value = 1 }
			}
			else = {
				change_variable = { which = emulation_years value = 1 }
			}
		}
		else_if = {
			limit = {
				OR = {
					check_variable = { which = emulation_progress value = 3 }
					AND = {
						has_lutheran_emulate_ambition = yes
						check_variable = { which = emulation_progress value = 2 }
					}
				}
			}
			character_event = { id = zarist.103 } # Complete success!
		}
		else_if = {
			limit = { NOT = { check_variable = { which = local_emulation_progress value = 0 } } }
			subtract_variable = { which = emulation_years value = 3 }
			if = {
				limit = { NOT = { check_variable = { which = emulation_years value = 0 } } }
				set_variable = { which = emulation_years value = 0 }
			}
			character_event = { id = zarist.102 } # Success!
		}
		else_if = {
			limit = { check_variable = { which = local_emulation_progress value = 1 } }
			subtract_variable = { which = emulation_years value = 2 }
			if = {
				limit = { NOT = { check_variable = { which = emulation_years value = 0 } } }
				set_variable = { which = emulation_years value = 0 }
			}
			character_event = { id = zarist.104 } # Fail!
		}
		else_if = {
			limit = {
				OR = {
					check_variable = { which = emulation_years value = 9 }
					AND = {
						check_variable = { which = emulation_years value = 7 }
						has_lutheran_emulate_ambition = yes
					}
				}
			}
			subtract_variable = { which = emulation_years value = 4 }
			set_character_flag = emulation_too_long
			character_event = { id = zarist.104 } # Fail!
		}
		else = {
			change_variable = { which = emulation_years value = 1 }
			character_event = { id = zarist.105 } # No progress
		}
	}
}
