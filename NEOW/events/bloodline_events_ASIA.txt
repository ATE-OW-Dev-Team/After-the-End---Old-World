##Form Gongheguo (Maoist)

narrative_event = {
	id = bludlines.1
	title = bludlines1Title
	desc = bludlines1Desc
	picture = GFX_evt_virgin_mary_icon

	major = yes
	show_ROOT = yes
	hide_from = yes
	
	only_playable = yes

	is_triggered_only = yes

	trigger = {
		NOT = {
			has_global_flag = gongheguo_refounded
		}
		religion = maoist
		primary_title = { title = e_maoist }
	}

	immediate = {
		if = {
            limit = {
                is_female = yes
                has_landed_title = e_maoist
            }
            create_bloodline = {
                type = chinese_emperor_female
                inheritance = matrilineal
            }
			new_bloodline = {
				set_description = "bloodline_desc_chinese"
			}
        }
        else_if = {
            limit = {
                is_female = no
                has_landed_title = e_maoist
            }
            create_bloodline = {
                type = chinese_emperor_male
            }
			new_bloodline = {
				set_description = "bloodline_desc_chinese"
			}
        }
		give_nickname = nick_the_glorious
		set_global_flag = gongheguo_refounded
		prestige = 750
		piety = 250
	}
	

	option = {
		name = bludlines1A
		trigger = {
			religion = maoist
		}
	}
	option = {
		name = bludlines1B
		trigger = {
			religion = red_cult
		}
	}
	option = {
		name = bludlines1C
		trigger = {
			OR = {
			religion = tridemism
			religion = taoist
			religion = white_lotus
			}
		}
	}
	option = {
		name = bludlines1D
		trigger = {
			religion = legalist
		}
	}
	option = {
		name = bludlines1E
		trigger = {
			OR = {
				religion = khanist
				religion = tengri_pagan_reformed
				religion = tengri_pagan
				}
			}
		}	
	option = {#None of the above
		name = bludlines1F
		trigger = {
			NOR = {
				religion = maoist
				religion = red_cult
				religion = taoist
				religion = white_lotus
				religion = qingzhen
				religion = xidaotang
				religion = tridemism
				religion = legalist
				religion = khanist
				religion = tengri_pagan_reformed
				religion = tengri_pagan
			}
		}
	}
}
