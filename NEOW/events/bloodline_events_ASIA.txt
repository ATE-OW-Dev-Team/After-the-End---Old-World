##Form Gongheguo (Maoist)

narrative_event = {
	id = bludlines.1
	title = bludlines1Title
	desc = bludlines1Desc
	picture = GFX_evt_maoist_virgin_mary_icon

	major = yes
	show_ROOT = yes
	hide_from = yes
	
	only_playable = yes

	is_triggered_only = yes

	trigger = {
		NOT = {
			has_global_flag = gongheguo_refounded
		}
		OR = { religion = maoist secret_religion = maoist }
		primary_title = { title = e_maoist }
	}

	immediate = {
		if = {
            limit = {
                is_female = yes
                has_landed_title = e_maoist
            }
            create_bloodline = {
                type = chinese_emperor_female
                inheritance = matrilineal
            }
			new_bloodline = {
				set_description = "bloodline_desc_chinese"
			}
        }
        else_if = {
            limit = {
                is_female = no
                has_landed_title = e_maoist
            }
            create_bloodline = {
                type = chinese_emperor_male
            }
			new_bloodline = {
				set_description = "bloodline_desc_chinese"
			}
        }
		give_nickname = nick_the_glorious
		set_global_flag = gongheguo_refounded
		prestige = 750
		piety = 250
	}
	
	weight_multiplier = {
		factor = 1
	}

	option = {
		name = bludlines1A
		trigger = {
			OR = { religion = maoist secret_religion = maoist }
		}
	}
	option = {
		name = bludlines1B
		trigger = {
			OR = { religion = red_cult secret_religion = red_cult }
		}
	}
	option = {
		name = bludlines1C
		trigger = {
			OR = {
			OR = { religion = tridemism secret_religion = tridemism }
			OR = { religion = taoist secret_religion = taoist }
			OR = { religion = white_lotus secret_religion = white_lotus }
			}
		}
	}
	option = {
		name = bludlines1D
		trigger = {
			OR = { religion = legalist secret_religion = legalist } 
		}
	}
	option = {
		name = bludlines1E
		trigger = {
			OR = {
				religion = khanist
				religion = tengri_pagan_reformed
				religion = tengri_pagan
				}
			}
		}	
	option = {#None of the above
		name = bludlines1F
		trigger = {
			NOR = {
				OR = { religion = maoist secret_religion = maoist }
				OR = { religion = red_cult secret_religion = red_cult }
				OR = { religion = taoist secret_religion = taoist }
				OR = { religion = white_lotus secret_religion = white_lotus }
				OR = { religion = qingzhen secret_religion = qingzhen }
				OR = { religion = xidaotang secret_religion = xidaotang }
				OR = { religion = tridemism secret_religion = tridemism }
				OR = { religion = legalist secret_religion = legalist } 
				religion = khanist
				religion = tengri_pagan_reformed
				religion = tengri_pagan
			}
		}
	}
}
