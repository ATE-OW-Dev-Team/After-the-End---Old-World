# A ruler is assimilated by the local culture
character_event = {
	id = 55010
	desc = EVTDESC55010
	picture = GFX_evt_stone_church

	is_triggered_only = yes # on_five_year_pulse, random_events, ruler_culture_conversion
	only_rulers = yes
	lacks_dlc = "Rajas of India" # Controlled by Decision: convert_to_local_culture
	min_age = 16
	only_capable = yes
	prisoner = no
	trigger = {
		NOT = { government = nomadic_government }
		primary_title = { 
			NOT = { conquest_culture = ROOT } 
			NOT = { culture = ROOT }
			any_previous_holder = {
				NOT = { character = ROOT } # The first holder of a title does not convert
			}
		}
		capital_scope = {
			NOT = { culture = ROOT }
		}
		top_liege = {
			OR = {
				character = ROOT
				ROOT = { 
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}
		NOT = { government = chinese_imperial_government }
	}

	weight_multiplier = {
		factor = 1

		mult_modifier = {
			factor = 2

			independent = no
			top_liege = {
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}

		mult_modifier = {
			factor = 2
			primary_title = {
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}

		mult_modifier = {
			factor = 0.5
			culture_group = capital_scope # Same culture group: slower assimilation
		}
	}

	option = {
		name = EVTOPTA55010

		# Piss off the appropriate vassals
		any_vassal = {
			show_scope_change = no

			limit = {
				culture = ROOT
				is_adult = yes
			}
			opinion = {
				name = opinion_adopted_foreign_ways
				who = ROOT
				years = 10
			}

			character_event = { id = 55011 }
		}

		culture = capital_scope
	}
	option = {
		name = EVTOPTB55010

		ai_chance = { factor = 0 }
	}
}
