title_vestalis = {
	dignity = 0.9
	grant_limit = 5
	opinion_effect = 5

	monthly_salary = 0.02
	monthly_prestige = 0.5
	monthly_piety = 0.1

	allowed_to_hold = {
		is_female = yes
		is_landed = no
		is_betrothed = no
		is_married = no
		is_consort = no
		prisoner = no
		religion_openly_hellenic_or_reformed_trigger = yes
		OR = {
			has_minor_title = title_vestalis
			AND = {
				age = 6
				NOT = { age = 10 }
				NOT = { trait = bastard }
				NOT = { trait = legit_bastard }
				NOT = { trait = child_of_consort }
				father_even_if_dead = { religion_openly_hellenic_or_reformed_trigger = yes }
				mother = { religion_openly_hellenic_or_reformed_trigger = yes }
			}
		}
	}
		
	allowed_to_grant = {
		has_landed_title = k_hellenic_pagan
	}

	gain_effect = {
		remove_trait = hedonist
		add_trait = celibate
		add_trait = vestalis
		recalc_succession = yes
		set_character_flag = flag_vestalis
		k_hellenic_pagan = { owner = { change_variable = { which = "num_vestales" value = 1 } } }
	}
	lose_effect = {
	}

	message = yes
}
