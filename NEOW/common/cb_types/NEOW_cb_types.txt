# The special CB for the Rise of Jacobinism 
jacobin_rising = {
	name = CB_NAME_JACOBIN_RISING
	war_name = WAR_NAME_JACOBIN_RISING
	sprite = 8
	truce_days = 365
	hostile_against_others = no
	is_permanent = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	apply_short_occ_mod = no # Do not apply the 'recently_conquered' modifier to Holdings

	can_use = {
		ROOT = {
			has_character_flag = jacobinist_leader
		}
	}
	
	is_valid_title = {
		ROOT = {
			has_character_flag = jacobinist_leader
		}
		FROM = {
			any_realm_title = {
				de_jure_liege_or_above = PREVPREV
			}
		}
		d_jacobinist = {
			has_holder = no
		}
	}
	
	on_add = {
		general_on_add_effect = yes
	}

	on_success = {
		clr_global_flag = jacobin_revolt_ongoing
		set_global_flag = jacobin_revolt_successful
		clr_global_flag = jacobinist_revolution_ended
		set_global_flag = jacobinist_revolution_ended
		ROOT = {
			piety = 500
			religion_authority = {
				modifier = rise_of_jacobinism
			}
			achievement_liberte_done_effect = yes
			clr_character_flag = jacobinist_leader
		}
		FROM = {
			piety = -500
			hidden_tooltip = {
				any_courtier = { #Sacrifice part of the court.
					limit = {
						is_ruler = no
					}
					random = { 
						chance = 25 
						modifier = {
							factor = 3
							is_unimportant_trigger = yes
						}
						modifier = {
							factor = 3
							has_minor_title = title_commander
						}
						modifier = { #Less likely to be in battle.
							factor = 0.5
							is_adult = no
						}
						modifier = { #Less likely to be in battle.
							factor = 0.5
							is_adult = no
							is_child_of = FROM
						}
						death = { 
							death_reason = death_jacobinist_sacrifice 
							killer = ROOT
						}
					}
				}
				opinion = {
					modifier = opinion_abducted_peasants
					who = ROOT
				}
				any_vassal = {
					opinion = {
						modifier = opinion_abducted_peasants 
						who = ROOT
					}
				}				
				any_demesne_province = {
					add_province_modifier = { name = bloodthirsty_flower_war_province years = 10 stacking = yes } 
					change_variable = { which = prosperity_value value = -40 }
					if = { 
						limit = { NOT = { check_variable = { which = prosperity_value value = 0 } } } 
						set_variable = { which = prosperity_value value = 0 }
					}
				}
				any_vassal = {
					limit = {
						higher_tier_than = BARON
					}
					any_demesne_province = { 
						add_province_modifier = { name = bloodthirsty_flower_war_province years = 10 stacking = yes } 
						change_variable = { which = prosperity_value value = -40 } 
						if = { 
							limit = { NOT = { check_variable = { which = prosperity_value value = 0 } } } 
							set_variable = { which = prosperity_value value = 0 }
						}
					}
				}
				jacobinist_sacrifice_after_death_effect = yes
			}
		}
	}
	
	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {

				FROM = {
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
					random_realm_province = {
						limit = {
							de_jure_liege_or_above = PREVPREV
							NOT = { religion = ROOT }
						}
						religion = ROOT
					}
				}
				ROOT = {
					occupy_minors_of_occupied_settlements = FROM
					gain_all_occupied_titles = FROM
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
					}
					
					any_demesne_title = {
						limit = { tier = BARON }
						remove_holding_modifier = recently_conquered
						refill_holding_levy = yes
					}
					usurp_title = { target = PREV type = invasion }				
					d_jacobinist = {#tried moving this beneath the usurp kingdom line to stop the theocracy issue
						usurp_title = { target = ROOT type = invasion }
					}
				}
				
				if = {
					limit = {
						FROM = { is_liege_or_above = ROOT }
					}
					FROM = { imprison = ROOT }
				}
			}
		}
	}

	on_fail = {
		clr_global_flag = jacobin_revolt_ongoing
		clr_global_flag = jacobinist_revolution_ended
		set_global_flag = jacobinist_revolution_ended
		FROM = {
			piety = 250
		}
		ROOT = {
			piety = -250
			clr_character_flag = jacobinist_leader
		}
	}

	on_reverse_demand = {
		clr_global_flag = jacobin_revolt_ongoing
		set_global_flag = jacobins_stopped
		clr_global_flag = jacobinist_revolution_ended
		set_global_flag = jacobinist_revolution_ended
		ROOT = {
			piety = -500
			clr_character_flag = jacobinist_leader
			remove_character_modifier = jacobinist_revolution_leader
			clear_persistent_event_target = jacobinist_revolution_target
			set_character_flag = jacobinist_failed_revolution
			imprison = FROM
		}
		FROM = {
			piety = 250
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_piety = 250
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_piety = 250 }
		}
		activate_title = { title = d_jacobinist status = no }
	}
	
	on_attacker_leader_death = {
		clr_global_flag = jacobin_revolt_ongoing
		clr_global_flag = jacobinist_revolution_ended
		set_global_flag = jacobinist_revolution_ended
		ROOT = {
			clr_character_flag = jacobinist_leader
		}
		end_war = invalid
		activate_title = { title = d_jacobinist status = no }
	}

	attacker_ai_victory_worth = {
		factor = 200
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
