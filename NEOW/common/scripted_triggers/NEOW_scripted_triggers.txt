			is_female = no

		}
		AND = {
			is_female = yes 
			liege = { has_religion_feature = religion_matriarchal }
		}
		AND = {
			is_female = yes 
			is_feminist_religion = yes
			liege = { is_feminist_religion = yes }
		}

			AND = { 
				is_female = yes

				name = gender
				value = all

		}
	}
