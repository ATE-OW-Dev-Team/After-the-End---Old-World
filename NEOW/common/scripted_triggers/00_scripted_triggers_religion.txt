ROOTs_religion_is_playable_trigger = {
	ROOT = {
		show_scope_change = no

		dlc_religion_check_trigger = yes
	}
}

ROOTs_secret_religion_is_playable_trigger = { 
	# To block players from getting a game over if they don't own all DLCs
	ROOT = {
		show_scope_change = no

		trigger_if = {
			limit = { is_nomadic = no }
			trigger_if = {
				limit = {
					secret_religion_group  = jewish_group
					NOT = { has_dlc = "Sons of Abraham" }
				}
				has_dlc = "Sons of Abraham"
			}
			trigger_else_if = {
				limit = {
					secret_religion_group  = muslim
					NOT = { has_dlc = "The Sword of Islam" }
				}
				has_dlc = "The Sword of Islam"
			}
			trigger_else_if = {
				limit = {
					OR = {
						secret_religion_group  = indian_group
						secret_religion_group = buddhist_group
					}
					NOT = { has_dlc = "Rajas of India" }
				}
				has_dlc = "Rajas of India"
			}
			trigger_else_if = {
				limit = {
					secret_religion_group  = pagan_group
					NOR = { 
						has_dlc = "The Old Gods" 
						has_dlc = "Holy Fury"
					}
				}
				OR = {
					has_dlc = "The Old Gods" 
					has_dlc = "Holy Fury" 
				}
			}
			trigger_else_if = {
				limit = {
					secret_religion_group  = pagan_group
					NOT = { has_dlc = "The Old Gods" }
				}
				has_dlc = "The Old Gods"
			}
		}
	}
}

		}

		limit = { PREV = { true_religion_group_muslim_trigger = yes } }
		OR = {
			trait = sympathy_islam

			trait = bhagavanist_akbar
			trait = muwa_sunni
			trait = muwa_shiite

	}

		limit = { PREV = { true_religion_group_pagan_group_trigger = yes } }
		OR = {
			trait = sympathy_pagans
			trait = vaishna_ganda_hindu
			trait = bhagavanist_akbar
			trait = blorg_tolerations
			trait = heroic_tales
		}

				religion_group = zoroastrian_group
			}
			NOT = { has_dlc = "The Old Gods" }
		}
		trigger_else = {
			always = no
		}
	}
	trigger_else = {
		is_nomadic = no
	}
}

dlc_religion_check_trigger = {
	trigger_if = {
		limit = { is_nomadic = no }
		dlc_religion_check_trigger_without_nomad_check = yes
	}
}

dlc_religion_check_trigger_without_nomad_check = {
	trigger_if = {
		limit = {
			religion_group = jewish_group
		}
		has_dlc = "Sons of Abraham"
	}
	trigger_else_if = {
		limit = {
			religion_group = muslim
		}
		has_dlc = "The Sword of Islam"
	}
	trigger_else_if = {
		limit = {
			OR = {
				religion_group = buddhist_group
				religion_group  = indian_group
			}
		}
		has_dlc = "Rajas of India"
	}
	trigger_else_if = {
		limit = {
			religion_group = pagan_group
		}
		OR = {
			has_dlc = "The Old Gods"
			has_dlc = "Holy Fury"
		}
	}
	trigger_else_if = {
		limit = {
			religion_group = zoroastrian_group
		}
		has_dlc = "The Old Gods"
	}
}
