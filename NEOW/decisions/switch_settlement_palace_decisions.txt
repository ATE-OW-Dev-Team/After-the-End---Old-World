settlement_decisions = {

	convert_settlement_castle = { 			# Convert holding to CASTLE
		filter = owned
		ai_target_filter = owned
		ai_check_interval = 30
		
		from_potential = {
			is_tribal = no
			is_nomadic = no
			independent = yes
			OR = {
				government = oligarchic_republic
				government = roman_imperial_government
			}
		}
		
		potential = {
			holding_type = family_palace
			holder_scope = {
				character = FROM
			}
		}
		
		allow = {
			holder = FROM
			FROM = {
				wealth = 250
			}
			AND = {
				holding_type = FAMILY_PALACE
				location = {
					any_province_holding = {
						is_capital = no
						holding_type = FAMILY_PALACE
					}
				}
			}
		}
		
		effect = {
			if = {
				limit = {
					holding_type = FAMILY_PALACE
				}
				FROM = {
					hidden_tooltip = {
						any_vassal = {
							limit = {
								OR = {
									is_republic = yes
									government = oligarchic_republic
									government = roman_imperial_government
								}
							}
							opinion = {
							modifier = opinion_angry
							who = FROM
							years = 10
							}
						}
					}
					custom_tooltip = { text = republic_vassal_angry }
				}
			}
			FROM = {
				wealth = -250
			}
			convert_to = CASTLE
		}
		ai_will_do = {
			factor = 1 # On average ca 1 year before taken
			modifier = {
				factor = 0.01 # Modified down to ca 10 year average.
			}
		}
	}
	convert_settlement_family_palace = {				
		filter = owned
		ai_target_filter = owned
		ai_check_interval = 60
		
		from_potential = {
			is_tribal = no
			is_nomadic = no
			independent = yes
			OR = {
				government = oligarchic_republic
				government = roman_imperial_government
			}
		}
		
		potential = {
			holding_type = castle
			holder_scope = {
				character = FROM
			}
		}
		
		allow = {
			holder = FROM
			FROM = {
				wealth = 1250
			}
			AND = {
				holding_type = castle
				location = {
					any_province_holding = {
						is_capital = no
						holding_type = castle
					}
				}
			}
		}
		
		effect = {
			if = {
				limit = {
					holding_type = castle
				}
				FROM = {
					hidden_tooltip = {
						any_vassal = {
							limit = {
								is_feudal = yes
								NOR = {
									government = oligarchic_republic
									government = roman_imperial_government
								}
							}
							opinion = {
							modifier = opinion_angry
							who = FROM
							years = 10
							}
						}
					}
					custom_tooltip = { text = feudal_vassal_angry }
				}
			}
			FROM = {
				wealth = -1250
			}
			convert_to = FAMILY_PALACE
		}
		ai_will_do = {
			factor = 1 # On average ca 1 year before taken
			modifier = {
				factor = 0.01 # Modified down to ca 10 year average.
			}
		}
	}

}
