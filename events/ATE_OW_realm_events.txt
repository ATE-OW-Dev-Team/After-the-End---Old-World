province_event = {
	id = ATEOW_realm_event.1000
	hide_window = yes
	trigger = {
		owner = {
			top_liege = {
				culture = PREV
				has_landed_title = e_austria_hungary
				OR = {
					AND = {
						culture = austrian
						ROOT = { culture = hungarian }
					}
					AND = {
						culture = hungarian
						ROOT = { culture = austrian }
					}
				}
			}
		}
		NOT = { has_province_modifier = accepted_culture }
	}
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
		add_province_modifier = { name = accepted_culture duration = -1 }
	}
}

character_event = {
	id = ATE_WWW_DEBUG.3
	desc = "DEBUG"
	picture = GFX_evt_capitol_building
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			ai = no 
			AND = {
				has_game_rule = {
					name = supernatural_events
					value = unrestricted
				}
				has_game_rule = {
					name = silly_events
					value = unrestricted
				}
			}
		}
		NOT = {
			has_game_rule = {
				name = supernatural_events
				value = off
			}
		}
		NOT = {
			has_game_rule = {
				name = silly_events
				value = off
			}
		}
	}

	option = {
		if = {
			limit = { ai = no }
			set_global_flag = bear_player
			random_playable_ruler = {
				limit = { 
					is_landed = yes
					is_playable = yes
					is_feudal = yes 
					government = feudal_government
				}
				set_player_character = THIS
			}
		}
		c_jan_mayen = {
			location = { culture = bear }
		}
		culture = bear
		set_graphical_culture = bear
		any_realm_character = {
			culture = bear
			set_graphical_culture = bear
			culture = bear
		}
		create_character = {
			random_traits = yes
			culture = bear
			race = bear
			female = no
			religion = lutheran
			dynasty = ROOT
			trait = stubborn
			trait = fair
			trait = strong
			martial = 5
			diplomacy = 1
			stewardship = 7
			intrigue = 7
			learning = 8
			flag = historical_character
		}
		new_character = {
			set_name = "Bjorn"
			ROOT = {
				any_child_even_if_dead = {
					set_father = PREVPREV
				}
				any_spouse = {
					add_spouse = PREVPREV
				}
				abdicate_to = PREV
			}
			if = {
				limit = { has_global_flag = bear_player }
				set_player_character = THIS
			}
		}
		death = yes
		culture = bear
		clr_global_flag = bear_player
	}
}
