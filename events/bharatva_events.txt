# Formation of the Shiv Sena through decision
narrative_event = {
	id = bharatva.1
	is_triggered_only = yes
	title = bharatva1Title
	picture = GFX_evt_roi_elephants_in_battle
	border = GFX_event_narrative_frame_religion

	major = yes	
	
	desc = {
		text = bharatva1Desc
	}

	immediate = {
		activate_title = { title = d_shiv_sena status = yes }
		create_random_soldier = {
			dynasty = none
			random_traits = yes
			add_trait = zealous
			religion = bharatva
			culture = hindustani
		}
		new_character = {

			d_shiv_sena = {
				grant_title = PREV
			}
			wealth = 500
		}
		
		religion_authority = {#if this doesn't work, try moving it to inside the new_character block, this command is valid from the scope of a character
			modifier = religious_order_formed
			years = 10
		}		
	}
	
	option = {#That's good
		name = EVTOPTA_bharatva1
		trigger = {
			religion = bharatva
		}
	}
	option = {#We really hate them
		name = EVTOPTB_bharatva1
		trigger = {
			OR = {
				religion = neovedantic
				religion = naxalite
				religion = naxalite_reformed
			}
		}
	}
	option = {#We really hate them too
		name = EVTOPTC_bharatva1
		trigger = {
			religion_group = muslim
		}
	}
	option = {#They're a bit overzealous
		name = EVTOPTD_bharatva1
		trigger = {
			OR = {
				AND = {
					religion_group = indian_group
					NOR = {
						religion = bharatva
						religion = neovedantic
					}
				}
				religion_group = buddhist_group
			}
		}
	}	
	option = {#Who cares?
		name = EVTOPTE_bharatva1
		trigger = {
			NOR = {
				religion_group = indian_group
				religion_group = buddhist_group
				religion_group = muslim
				religion = naxalite
				religion = naxalite_reformed
			}
		}
	}
}
