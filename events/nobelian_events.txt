		text = nobelian1Desc
	}

				}
		}
		}
	}

		name = EVTOPTB_nobelian1

	}
	option = {#Third option for if you have a mentor
		name = EVTOPTC_nobelian1

				any_courtier = {
					has_character_flag = is_nobelian_mentor

	}
}

character_event = {#You fail - there's still a chance that you'll get a Mystic Insight, but you'll suffer mental damage even if you do
	id = nobelian.7
	is_triggered_only = yes
	picture = GFX_evt_book_carolingian_script
	border = GFX_event_normal_frame_religion

	desc = {
		text = nobelian7Desc
	}
	
	option = {#
		name = EVTOPTA_nobelian7
		random_list = {
			50 = {
				add_artifact = mystic_insight

			}
			50 = {
			
			}
		}
		if = {
			limit = {
				NOR = {
					trait = stressed
					trait = depressed
					trait = lunatic
					trait = possessed
				}
				trait = stressed
			}
		}
		else_if = {
			limit = {
				trait = stressed
				NOR = {
					trait = depressed
					trait = lunatic
					trait = possessed
				}
			}
			remove_trait = stressed
			trait = depressed
		}
		else_if = {
			limit = {
				trait = depressed
				NOR = {
					trait = lunatic
					trait = possessed
				}
			}
			trait = lunatic
		}
		else_if = {
			limit = {
				trait = lunatic
				NOT = {
					trait = possessed
				}
			}
			trait = possessed
		}
		else_if = {
			limit = {
				trait = stressed
				trait = depressed
				trait = lunatic
				trait = possessed
			}
			random_list = {
				50 = {
					death = { death_reason = death_vanished }
				}
				50 = {
					trait = severely_injured
				}
			}
		}
	}
}
