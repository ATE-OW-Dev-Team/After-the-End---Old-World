character_event = {
	id = nobelian.1
	is_triggered_only = yes
	picture = GFX_evt_book_carolingian_script
	border = GFX_event_normal_frame_religion

	desc = {
		text = nobelian1Desc
	}
	
	option = {#Search for a Mentor
		name = EVTOPTA_nobelian1
		trigger = {
			NOT = {
				any_courtier = {
					has_character_flag = is_nobelian_mentor
				}				
			}
		}

		random_list = {

				33 = {#1/3 chance he asks for an unspecified future payment (bad events may happen to you because of this, later)

				}
				33 = {#1/3 chance he asks for a pound of flesh
					character_event = { id = nobelian.4 days = 35 }#You look for a mentor
				}
		}
		}
	}
	
	option = {#Do it myself
		name = EVTOPTB_nobelian1
		character_event = { id = nobelian.5 days = 7 }
	}
	option = {#Third option for if you have a mentor
		name = EVTOPTC_nobelian1
		trigger = {
				any_courtier = {
					has_character_flag = is_nobelian_mentor
				}				
		}
		character_event = { id = nobelian.8 days = 7 }
	}
}

character_event = {#You fail - there's still a chance that you'll get a Mystic Insight, but you'll suffer mental damage even if you do
	id = nobelian.7
	is_triggered_only = yes
	picture = GFX_evt_book_carolingian_script
	border = GFX_event_normal_frame_religion

	desc = {
		text = nobelian7Desc
	}
	
	option = {#
		name = EVTOPTA_nobelian7
		random_list = {
			50 = {
				add_artifact = mystic_insight
				new_artifact = { set_creation_date = 1.1.1 }
			}
			50 = {
			
			}
		}
		if = {
			limit = {
				NOR = {
					trait = stressed
					trait = depressed
					trait = lunatic
					trait = possessed
				}
				trait = stressed
			}
		}
		else_if = {
			limit = {
				trait = stressed
				NOR = {
					trait = depressed
					trait = lunatic
					trait = possessed
				}
			}
			remove_trait = stressed
			trait = depressed
		}
		else_if = {
			limit = {
				trait = depressed
				NOR = {
					trait = lunatic
					trait = possessed
				}
			}
			trait = lunatic
		}
		else_if = {
			limit = {
				trait = lunatic
				NOT = {
					trait = possessed
				}
			}
			trait = possessed
		}
		else_if = {
			limit = {
				trait = stressed
				trait = depressed
				trait = lunatic
				trait = possessed
			}
			random_list = {
				50 = {
					death = { death_reason = death_vanished }
				}
				50 = {
					trait = severely_injured
				}
			}
		}
	}
}
