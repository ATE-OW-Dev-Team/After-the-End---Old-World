character_event = {
	id = COURTE.1
	desc = EVTDESC_COURTE_1
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	prisoner = no
	only_playable = yes
	capable_only = yes
	only_independent = yes
	
	trigger = {
		has_dlc = "Sons of Abraham"
		capital_scope = {
			region = custom_israeli_region
		}
		NOT = { has_character_modifier = expelled_jewish }
		NOT = { trait = incapable }
		NOT = { is_inaccessible_trigger = yes }
	}
	
	mean_time_to_happen = {
		months = 1500
		modifier = {
			factor = 0.95
			religion = jewish 
		}
		modifier = {
			factor = 0.95
			culture = israeli 
		}
	}
	
	immediate = {
		create_character = {
			random_traits = yes
			culture = israeli 
			dynasty = culture
			religion = jewish
			female = no
			age = 30
			trait = charismatic_negotiator
			attributes = {
				diplomacy = 10
			}
			flag = ai_flag_refuse_conversion
		}
		new_character = {
			remove_lifestyle_trait_effect = yes
			random_list = {
				70 = {
					add_trait = socializer
				}
				30 = {
					add_trait = socializer
					add_trait = shrewd
				}
			}
			save_event_target_as = jewish_courtier
		}
	}
	
	option = {
		name = EVTOPTA_SoA_200
		custom_tooltip = {
			text = jewish_courtier_tooltip
		}
	}
}
