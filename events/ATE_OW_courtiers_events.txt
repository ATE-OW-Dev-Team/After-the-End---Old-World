character_event = {
	id = COURTE.1
	desc = EVTDESC_COURTE_1
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	prisoner = no
	only_playable = yes
	capable_only = yes
	only_independent = yes
	
	trigger = {
		has_dlc = "Sons of Abraham"
		OR = {
			religion_group = christian
			religion_group = muslim
			religion_group = zoroastrian_group
			religion_group = jewish_group
			religion_group = old_world_cultist_group 
			religion_group = esoteric 
			religion_group = indian_group 
			religion_group = buddhist_group 
			culture_group = chinese_group
			culture_group = korean_group
		}
		NOT = { has_character_modifier = expelled_jewish }
		NOT = { trait = incapable }
		NOT = { is_inaccessible_trigger = yes }
	}
	
	mean_time_to_happen = {
		months = 1100
	}
	
	immediate = {
		create_character = {
			random_traits = yes
			culture = israeli 
			dynasty = culture
			religion = jewish
			female = no
			age = 30
			trait = charismatic_negotiator
			attributes = {
				diplomacy = 10
			}
			flag = ai_flag_refuse_conversion
		}
		new_character = {
			remove_trait = socializer
			remove_trait = gamer
			remove_trait = hedonist
			remove_trait = duelist
			remove_trait = hunter
			remove_trait = strategist
			remove_trait = administrator
			remove_trait = architect
			remove_trait = gardener
			remove_trait = schemer
			remove_trait = impaler
			remove_trait = seductress
			remove_trait = scholar
			remove_trait = theologian
			remove_trait = mystic
			random_list = {
				70 = {
					add_trait = socializer
				}
				30 = {
					add_trait = socializer
					add_trait = shrewd
				}
			}
			save_event_target_as = jewish_courtier
		}
	}
	
	option = {
		name = EVTOPTA_SoA_200
		custom_tooltip = {
			text = jewish_courtier_tooltip
		}
	}
}
