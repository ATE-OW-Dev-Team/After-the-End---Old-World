#Adessite
character_event = {
	id = MPU.1
	desc = "EVTDESCMPU.1"
	picture = GFX_evt_frankisc
	
	only_playable = yes
	
	trigger = {
		OR = {
			culture = wallachian
			culture = transylvanian
			culture = moldavian
		}
		any_demesne_province = {
			OR = {
				culture = ukrainer
			}
			any_province_holding = {
				NOT = { holding_type = nomad }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.5
			culture = adessite
		}
		modifier = {
			factor = 0.75
			NOT = { learning = 7 }
			learning = 5
		}
		modifier = {
			factor = 0.5
			NOT = { learning = 11 }
			learning = 7
		}
		modifier = {
			factor = 0.25
			learning = 15
		}
		modifier = {
			factor = 1
			NOT = { learning = 5 }
			learning = 1
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 1 }
		}
		modifier = {
			factor = 0.65
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTAMPU.1
		random_demesne_province = {
			limit = {
				OR = {
					culture = ukrainer
				}
				any_province_holding = {
					NOT = { holding_type = nomad }
				}
			}
			culture = adessite
		}
	}
}
